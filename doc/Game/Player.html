def self.from_json(data)
  new(
    xp: data['xp'],
    level: data['level'],
    hr: data['hr'],
    zenny: data['zenny'],
    max_hp: data['max_hp'],
    hp: data['hp'],
    time: data['time'],
    inventory: data['inventory'],
    death_time: data['death_time'] || Time.new('2017')
  )
end
